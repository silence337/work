$(function($){
//SOME CODE
    windowResize();
    readTableJson();
    
    $('span.refreshTable').on('click',function(e){
        $('.tableBodyRow').remove();
        readTableJson();
    });
}(jQuery));
 
function windowResize(){
    $(window).on('resize load',function(){
        var $headerHeight = $('header').height();
        $('section .MainEmptyDiv').css('height',$headerHeight);
    });
}
 
function readTableJson(){
    $.ajax({
        'type':'GET',
        'url':'ArtistList.json',
        'dataType':'json',
        success:function(jsonFile){
            var SongTable = '';
            for(var i=0;i<jsonFile.length;i++){
                SongTable += '<tr class="tableBodyRow"><td class="artist"><figure><img alt="' + jsonFile[i].name +'"src="'+jsonFile[i].image+'"><figcaption>'+jsonFile[i].name+ '</figcaption></figure></td>';
                SongTable += '<td class="songName">'+jsonFile[i].song+'</td>';
                SongTable += '<td class="buyVote"><span title="buy" class="buy"></span><span title="book" class="book"></span><span title="like" class="voteMusic"></span></td>';
                SongTable += '</tr>';
            }
            $('table').append(SongTable);
        }
    });
}
